Code to open all excel files in a folder

  Sub OpenFiles()
Dim ThisSheet, Othersheet As Workbook
Application.ScreenUpdating = False

Set ThisSheet = ActiveWorkbook 		'this sheet is the excel file where the macro is

Dim MyFolder As String
Dim MyFile As String

MyFolder = ThisSheet.Path & "\FOLDER NAME\"           'full path to folder where files live - needs a \ at the end - folder needs to be in same place as ThisSheet

MyFile = Dir(MyFolder & "*.xl*")                                'list all excel files in folder


Do While MyFile <> ""                                          'open each file in folder until none left
Set Othersheet = Workbooks.Open(MyFolder & MyFile)             'Othersheet is prison return file

Othersheet.Close SaveChanges:=False
MyFile = Dir

Loop

End Sub
